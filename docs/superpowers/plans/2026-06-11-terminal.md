# 交互式终端实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development 或 executing-plans 逐任务实现。步骤用 `- [ ]` 跟踪。

**Goal:** 工作区里一个 Ghostty 风悬浮终端窗口,经三跳双向流(WS→backend→worker→sandbox PTY)直连用户沙箱的交互式 bash,能跑 vim/top/Ctrl-C/颜色。

**Architecture:** 见 spec `docs/superpowers/specs/2026-06-11-terminal-design.md`。三跳双向流,拓扑同 RunTurn:backend 不直连沙箱,经 gRPC metadata 把 sandbox endpoint+token 传给 worker,worker 透传桥。临时 PTY、软状态持久(HISTFILE+cwd)、保活第二档、单终端。

**Tech Stack:** Python(sandbox PTY via stdlib `pty`/`os`/`fcntl`/`termios`、worker/backend gRPC aio 双向流、FastAPI WebSocket)+ React/xterm.js。

**关键事实:**
- proto 生成:仓库根跑 `bash scripts/gen_protos.sh` → 落 `packages/common/src/agent_cloud/v1/`。两 proto 同 `package agent_cloud.v1`,worker.proto `import` sandbox.proto 复用 Terminal 消息。
- sandbox token 校验:`x-sandbox-token` metadata + `hmac.compare_digest`(server.py 既有,新增 RPC 必须复制)。
- backend↔worker:sandbox 连接信息走 gRPC **metadata**(`x-sandbox-endpoint`/`x-sandbox-token`),不进消息体。
- WS 鉴权:浏览器 WS 不能带 header → token 走 subprotocol `["bearer", <access>]`。
- 测试:sandbox/worker `cd <svc> && uv run pytest`;backend `cd services/backend && TESTCONTAINERS_RYUK_DISABLED=true uv run pytest -m "not docker"`;前端 `cd frontend && npm run lint && npx vitest run`。

- [ ] **实现** `pty_session.py`:
```python
from __future__ import annotations
import asyncio, fcntl, os, pty, signal, struct, termios
from pathlib import Path

class PtySession:
    """一个 PTY + 交互式 bash 子进程。read 经 loop.add_reader 异步化。"""
    def __init__(self, workdir: Path, rows: int = 24, cols: int = 80, env: dict | None = None):
        self._workdir = Path(workdir)
        self._rows, self._cols = rows or 24, cols or 80
        self._env = env
        self._master: int | None = None
        self._proc: asyncio.subprocess.Process | None = None
        self._queue: asyncio.Queue[bytes] = asyncio.Queue()
        self._closed = False

    async def start(self) -> None:
        self._workdir.mkdir(parents=True, exist_ok=True)
        master, slave = pty.openpty()
        self._set_winsize(master, self._rows, self._cols)
        env = self._env if self._env is not None else dict(os.environ)
        env.setdefault("TERM", "xterm-256color")
        self._proc = await asyncio.create_subprocess_exec(
            "bash", "-i",
            stdin=slave, stdout=slave, stderr=slave,
            cwd=str(self._workdir), env=env,
            start_new_session=True,  # setsid:独立进程组,Ctrl-C 作用于前台组而非整服务
            preexec_fn=_make_controlling_tty(slave),
        )
        os.close(slave)
        os.set_blocking(master, False)
        self._master = master
        asyncio.get_running_loop().add_reader(master, self._on_readable)

    def _on_readable(self) -> None:
        try:
            data = os.read(self._master, 65536)
        except (BlockingIOError, InterruptedError):
            return
        except OSError:
            data = b""  # PTY 关闭(子进程退出)
        if data:
            self._queue.put_nowait(data)
        else:
            asyncio.get_running_loop().remove_reader(self._master)
            self._queue.put_nowait(b"")  # EOF 哨兵

    async def read(self) -> bytes:
        return await self._queue.get()

    async def write(self, data: bytes) -> None:
        if self._master is not None and not self._closed:
            os.write(self._master, data)

    def resize(self, rows: int, cols: int) -> None:
        if self._master is not None:
            self._set_winsize(self._master, rows or 24, cols or 80)

    @staticmethod
    def _set_winsize(fd: int, rows: int, cols: int) -> None:
        fcntl.ioctl(fd, termios.TIOCSWINSZ, struct.pack("HHHH", rows, cols, 0, 0))

    async def wait(self) -> int:
        assert self._proc is not None
        return await self._proc.wait()

    async def close(self) -> None:
        if self._closed:
            return
        self._closed = True
        if self._master is not None:
            try:
                asyncio.get_running_loop().remove_reader(self._master)
            except Exception:
                pass
        if self._proc and self._proc.returncode is None:
            try:
                os.killpg(os.getpgid(self._proc.pid), signal.SIGKILL)
            except ProcessLookupError:
                pass
            try:
                async with asyncio.timeout(2):
                    await self._proc.wait()
            except TimeoutError:
                pass
        if self._master is not None:
            try:
                os.close(self._master)
            except OSError:
                pass


def _make_controlling_tty(slave: int):
    def _pre():
        fcntl.ioctl(slave, termios.TIOCSCTTY, 0)  # slave 成为 controlling tty(job control)
    return _pre
```
- [ ] **server.py** 加 `Terminal` handler(import `pty_session.PtySession`、`tools._resolve_within`、`asyncio`):
```python
async def Terminal(self, request_iterator, context):
    # ⚠️ 复制 ExecTool 的 token 校验(server.py 注释明确要求)
    if self._token:
        md = dict(context.invocation_metadata() or ())
        if not hmac.compare_digest(md.get("x-sandbox-token", ""), self._token):
            await context.abort(grpc.StatusCode.UNAUTHENTICATED, "invalid sandbox token")
    # 首帧必须是 start
    first = await request_iterator.read()  # aio: 用 async for 取首个;见实现注
    if first is grpc.aio.EOF or first.WhichOneof("msg") != "start":
        await context.abort(grpc.StatusCode.INVALID_ARGUMENT, "first message must be start")
    try:
        workdir = _resolve_within(self._base, first.start.work_subdir)
    except ValueError:
        await context.abort(grpc.StatusCode.INVALID_ARGUMENT, "invalid work_subdir")
    sess = PtySession(workdir, first.start.rows, first.start.cols)
    await sess.start()
    async def _pump_in():
        async for msg in request_iterator:
            kind = msg.WhichOneof("msg")
            if kind == "input":
                await sess.write(msg.input)
            elif kind == "resize":
                sess.resize(msg.resize.rows, msg.resize.cols)
    pump = asyncio.create_task(_pump_in())
    try:
        while True:
            data = await sess.read()
            if data == b"":  # EOF:子进程退出
                break
            yield sandbox_pb2.TerminalServerMsg(output=data)
        code = await sess.wait()
        yield sandbox_pb2.TerminalServerMsg(exit_code=code)
    finally:
        pump.cancel()
        await sess.close()
```
注:aio servicer 的 `request_iterator` 是 async iterator;取首帧用 `first = await request_iterator.__anext__()`(EOF 抛 StopAsyncIteration)。实现时用 try/except StopAsyncIteration 包裹;`_resolve_within` 从 `tools` import。`run_tool` 路径里 `_resolve_within` 是模块级函数,直接 `from agent_cloud_sandbox.tools import _resolve_within`。

```python
from fastapi import APIRouter, WebSocket
import grpc
from agent_cloud.v1 import sandbox_pb2, worker_pb2_grpc
from agent_cloud_backend.auth.security import decode_access_token
from agent_cloud_backend.config import get_settings
from agent_cloud_backend.db import get_sessionmaker
from agent_cloud_backend.repositories.user import UserRepository
from agent_cloud_backend.repositories.sandbox_registry import SandboxRegistryRepository
from agent_cloud_backend.sandbox.deps import get_sandbox_manager
import asyncio, json, uuid, time

router = APIRouter(tags=["terminal"])

def _token_from_subprotocols(ws: WebSocket) -> str | None:
    # 浏览器 WS 不能带 Authorization header → token 走 subprotocol ["bearer", <token>]
    raw = ws.headers.get("sec-websocket-protocol", "")
    parts = [p.strip() for p in raw.split(",")]
    if len(parts) >= 2 and parts[0] == "bearer":
        return parts[1]
    return None

@router.websocket("/terminal")
async def terminal_ws(ws: WebSocket):
    settings = get_settings()
    token = _token_from_subprotocols(ws)
    uid = decode_access_token(token, secret=settings.auth_secret) if token else None
    if uid is None:
        await ws.close(code=1008)
        return
    async with get_sessionmaker()() as db:
        user = await UserRepository(db).get(uuid.UUID(uid))
    if user is None:
        await ws.close(code=1008)
        return
    # accept:必须回显选中的 subprotocol "bearer"
    await ws.accept(subprotocol="bearer")
    manager = get_sandbox_manager()
    conn = await manager.get_endpoint_for_user(user.id)
    md = (("x-sandbox-endpoint", conn.endpoint), ("x-sandbox-token", conn.token))
    options = [("grpc.max_send_message_length", 16*1024*1024),
               ("grpc.max_receive_message_length", 16*1024*1024)]
    async with grpc.aio.insecure_channel(settings.worker_endpoint, options=options) as ch:
        stub = worker_pb2_grpc.WorkerStub(ch)
        call = stub.Terminal(metadata=md)
        # 首帧 start(rows/cols 后续 resize 修正)
        await call.write(sandbox_pb2.TerminalClientMsg(
            start=sandbox_pb2.TerminalStart(work_subdir=f"sessions/{user.id}", rows=24, cols=80)))
        last_touch = 0.0
        async def _ws_to_worker():
            nonlocal last_touch
            while True:
                msg = await ws.receive()
                if msg["type"] == "websocket.disconnect":
                    break
                if (b := msg.get("bytes")) is not None:
                    await call.write(sandbox_pb2.TerminalClientMsg(input=b))
                    now = time.monotonic()  # 续租第二档:仅输入触发,≥10s 节流
                    if now - last_touch >= 10:
                        last_touch = now
                        async with get_sessionmaker()() as db:
                            await SandboxRegistryRepository(db).touch_for_user(user.id); await db.commit()
                elif (t := msg.get("text")) is not None:
                    r = json.loads(t)  # resize: {"rows":N,"cols":M}
                    await call.write(sandbox_pb2.TerminalClientMsg(
                        resize=sandbox_pb2.TerminalResize(rows=r["rows"], cols=r["cols"])))
        async def _worker_to_ws():
            async for out in call:
                if out.WhichOneof("msg") == "output":
                    await ws.send_bytes(out.output)
                else:
                    await ws.close(code=1000); return
        a = asyncio.create_task(_ws_to_worker()); b = asyncio.create_task(_worker_to_ws())
        try:
            await asyncio.wait({a, b}, return_when=asyncio.FIRST_COMPLETED)
        finally:
            a.cancel(); b.cancel()
            try: await call.done_writing()
            except Exception: pass
```
